# ── CUDA / Blackwell specific dependencies ──────────────────────
# Inherits everything from requirements.txt
-r requirements.txt

# bitsandbytes — q4/q8 quantization (works on CUDA)
bitsandbytes

# torchao — FP8 quantization via TorchAoConfig (recommended for GB10)
# Uses PyTorch native float8_e4m3fn, no CUTLASS dependency
torchao

# compressed-tensors — required for loading pre-quantized models
# (e.g. RedHatAI/Voxtral-Small-24B-2507-FP8-dynamic)
compressed-tensors

# NVFP4 support (B200/B100 only — broken on GB10/SM120, see README)
# fp_quant is required by transformers FPQuantConfig
# qutlass provides the NVFP4 CUTLASS kernels (build from source)
#   see: https://github.com/IST-DASLab/QuTLASS

# ── Flash Attention 2 (optional but recommended) ────────────────
# Since 2026-04, Voxtral defaults to attn_implementation="flash_attention_2"
# on Ampere/Ada/Hopper CUDA GPUs (sm_80/86/89/90). If flash-attn is not
# available, core/transcribe.py transparently falls back to eager and logs
# a hint — so this dep is NOT required for correctness, only for speed
# (~1.5-2x on Voxtral inference).
#
# NOT installed by default for two reasons:
#   1. The wheel build is ~20 min on first install and needs a matching
#      torch+CUDA combo, which is fragile in generic CI.
#   2. The NGC PyTorch image used by Dockerfile.spark (Blackwell/GB10)
#      already ships with flash-attn baked in — but Blackwell uses eager
#      by default anyway (see core/transcribe.py for the reasoning).
#
# Enable locally or in Dockerfile.cuda with:
#   pip install flash-attn --no-build-isolation
# Or pin a pre-built wheel:
#   pip install flash-attn==2.8.3 --no-build-isolation
# flash-attn
